#include <stdio.h>

int main()
 {
    int a, b, c;
    int big, small, medium;
    scanf("%d %d %d", &a, &b, &c);
    if (a >= b && a >= c)
        big = a;
    else if (b >= a && b >= c)
        big = b;
    else
        big = c;
    if (a <= b && a <= c)
        small = a;
    else if (b <= a && b <= c)
        small = b;
    else
        small = c;
medium = a + b + c - big -small;
printf("%d %d %d\n", big, small, medium);
}
