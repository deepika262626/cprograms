#include <stdio.h>

int main()
{
    int size;
    printf("enter the number of element to store:\n");
    scanf("%d",&size);
    int a[size];
    for(int i=0;i<=size-1;i++)
    {
        printf("enetr the %d value\n",i++);
        scanf("%d",&a[i]);
    }
    int max=a[1];
    for(int i=1;i<=size-1;i++)
    {
        if(a[i]<max)
        {
            max=a[i];
        }
    }
    printf("max=%d",max);
}
