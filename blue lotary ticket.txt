#include <stdio.h>

int main()
 {
    int a, b, c;
    scanf("%d %d %d", &a, &b, &c);
    if (a + b == 10 || b + c == 10 || a + c == 10) 
{
        printf("10\n");
    }
    else if (a > 10 || b > 10 || c > 10) 
{
        printf("5\n");
    }
    else {
        printf("0\n");
   }
}
